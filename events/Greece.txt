# Ультиматум Французов
country_event = {
	id = greece.1
	title = greece.1.t
	desc = greece.1.d
	picture = GFX_report_event_greece_civil_war_soldiers
	
	fire_only_once = yes
	
	trigger = {
		has_global_flag = WWI_is_now
		tag = GRE
		has_war = no
		GRR = { exists = yes }
		date > 1917.05.01
	}
	
	immediate = {
		set_global_flag = French_ultimatum_to_GRE_fired
	}

	mean_time_to_happen = {
		days = 125
	}
	
	option = { # С меня хватит! Я отрекаюсь от престола...
		name = greece.1.a
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				GRR = { NOT = { controls_state = 731 } }
			}
		}
		GRR = {
			annex_country = { target = GRE transfer_troops = yes }
			set_capital = { state = 47 }
		}
	}
	option = { # Да пошли вы!
		name = greece.1.b
		ai_chance = {
			factor = 1
			modifier = {
				factor = 100
				GRR = { NOT = { controls_state = 731 } }
			}
		}
		add_war_support = -0.05
		GRR = { drop_cosmetic_tag = yes }
		declare_war_on = {
			target = GRR
			type = annex_everything
		}
		if = {
			limit = { GER = { is_faction_leader = yes } }
			GER = {
				add_to_faction = GRE
			}
		}
		if = {
			limit = { has_country_flag = AUH_help_to_king_GRE }
			if = {
				limit = {
					has_dlc = "Together for Victory"
				}
				AUH = {
					GRE = { subjecting_script_with_transfer_parties_semi_independent = yes }
				}
				else = {
					AUH = { puppet = GRE }
				}
			}
		}
	}
}

country_event = {
	id = greece.2
	title = greece.2.t
	desc = greece.2.d
	picture = GFX_report_event_greece_civil_war_soldiers
	
	fire_only_once = yes
	hidden = yes
	
	trigger = {
		has_capitulated = no
		OR = {
			tag = GRE
			tag = GRR
		}
		OR = {
			GRE = { exists = no }
			GRE = { has_capitulated = yes }
			GRR = { exists = no }
			GRR = { has_capitulated = yes }
		}
	}

	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = greece.2.a
		if = {
			limit = { GRE = { exists = yes has_capitulated = no } }
			every_state = {
				limit = {
					OR = {
						is_core_of = GRR
						is_claimed_by = GRR
					}
				}
				GRR = {
					remove_state_core = PREV
					remove_state_claim = PREV
				}
			}
			GRE = {
				annex_country = {
					target = GRR
					transfer_troops = yes
				}
			}
		}
		if = {
			limit = { GRR = { exists = yes has_capitulated = no } }
			every_state = {
				limit = {
					OR = {
						is_core_of = GRR
						is_claimed_by = GRR
					}
				}
				GRE = {
					remove_state_core = PREV
					remove_state_claim = PREV
				}
			}
			GRR = {
				annex_country = {
					target = GRE
					transfer_troops = yes
				}
			}
		}
		set_capital = { state = 47 }
	}
}
