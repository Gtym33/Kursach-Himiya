add_namespace = spain

# Rif War
country_event = {
	id = spain.1
	title = spain.1.t
	desc = spain.1.d
	picture = GFX_report_event_spain_civil_war_soldiers
	
	fire_only_once = yes
	
	trigger = {
		tag = SPR
		OR = {
			has_government = market_liberalism
			has_government = conservatism
			has_government = authoritarism
			has_government = despotism
		}
		RIF = { exists = no }
		date > 1920.1.1
	}

	mean_time_to_happen = {
		days = 523
		modifier = {
			factor = 0.5
			surrender_progress > 0.2
		}
		modifier = {
			factor = 0.7
			surrender_progress > 0.1
		}
		modifier = {
			factor = 0.7
			has_political_power < 0
		}
		modifier = {
			factor = 0.25
			has_army_manpower = { size < 40000 }
		}
	}
	
	option = { # Release
		name = spain.1.a
		trigger = { is_ai = no }
		release = RIF
		RIF = { load_oob = RIF_war }
		if = {
			limit = { has_government = market_liberalism }
			add_popularity = {
				ideology = market_liberalism
				popularity = -0.15
			}
		}
		if = {
			limit = { has_government = conservatism }
			add_popularity = {
				ideology = conservatism
				popularity = -0.15
			}
		}
		if = {
			limit = { has_government = authoritarism }
			add_popularity = {
				ideology = authoritarism
				popularity = -0.15
			}
		}
		if = {
			limit = { has_government = despotism }
			add_popularity = {
				ideology = despotism
				popularity = -0.15
			}
		}
		add_political_power = -50
	}
	option = { # Declare war
		name = spain.1.b
		release = RIF
		declare_war_on = {
			target = RIF
			type = annex_everything
		}
		RIF = { load_oob = RIF_war }
		#news_event
	}
}
