# Вопрос с бывшим правительством
country_event = {
	id = rcw_kcr.1
	title = rcw_kcr.1.t
	desc = rcw_kcr.1.d
	picture = GFX_report_event_rus_kornilov_goverment
	
	trigger = {
		tag = RUS
		OR = {
			has_government = authoritarism
			has_government = despotism
			has_government = fascism
			has_government = national_syndicalism
		}
		has_country_flag = general_kornilov
	}
	
	mean_time_to_happen = {
		days = 9
		modifier = {
			factor = 0.01
			OR = {
				has_completed_focus = RUS_compromise_with_democrats
				has_completed_focus = RUS_disperse_Petrograd_Soviet
			}
		}
	}
	
	fire_only_once = yes
	
	immediate = {
		set_country_flag = RUS_Korn_ev_1
	}
	
	option = {
		name = rcw_kcr.1.a #Пусть разбегается
		if = {
			limit = { NOT = { has_completed_focus = RUS_compromise_with_democrats } }
			add_to_variable = { var = RUS_Kornilov_power value = -1 }
		}
		ai_chance = {
			factor = 0
		}
		remove_ideas_with_trait = head_of_government
		remove_ideas_with_trait = foreign_minister
		remove_ideas_with_trait = economy_minister
		remove_ideas_with_trait = security_minister
		remove_ideas_with_trait = head_of_intel
		remove_ideas_with_trait = military_minister
		add_ideas = {
			Nikolay_Khomyakov_1
			Alexander_Guchkov_2
			Pyotr_Heyden_3
			Boris_Savinkov_natsind_4
			Alexander_Krasilnikov_5
		}
		set_country_flag = RUS_kornilov_russia_38_nothing_government
	}
	option = {
		name = rcw_kcr.1.b #Разогнать
		trigger = {
			check_variable = { RUS_force_uses > 0 }
			NOT = { has_completed_focus = RUS_compromise_with_democrats }
		}
		add_political_power = 25
		add_to_variable = { var = RUS_force_uses value = -1 }
		add_popularity = {
			ideology = social_liberalism
			popularity = -0.05
		}
		add_popularity = {
			ideology = market_liberalism
			popularity = -0.05
		}
		remove_ideas_with_trait = head_of_government
		remove_ideas_with_trait = foreign_minister
		remove_ideas_with_trait = economy_minister
		remove_ideas_with_trait = security_minister
		remove_ideas_with_trait = head_of_intel
		remove_ideas_with_trait = military_minister
		add_ideas = {
			Nikolay_Khomyakov_1
			Alexander_Guchkov_2
			Pyotr_Heyden_3
			Boris_Savinkov_natsind_4
			Alexander_Krasilnikov_5
		}
		set_country_flag = RUS_kornilov_russia_38_razgon_government
	}
	option = {
		name = rcw_kcr.1.c #Переформировать из Кадетов
		if = {
			limit = { market_liberalism > 0.2 }
			add_to_variable = { var = RUS_Kornilov_power value = 1 }
		}
		add_popularity = {
			ideology = social_liberalism
			popularity = -0.075
		}
		add_popularity = {
			ideology = market_liberalism
			popularity = 0.05
		}
		add_political_power = -25
		add_stability = 0.025
		remove_ideas_with_trait = head_of_government
		remove_ideas_with_trait = foreign_minister
		remove_ideas_with_trait = economy_minister
		remove_ideas_with_trait = security_minister
		remove_ideas_with_trait = head_of_intel
		remove_ideas_with_trait = military_minister
		add_ideas = {
			Nikolay_Khomyakov_1
			Pavel_Milyukov_2
			Alexander_Konovalov_3
			Boris_Savinkov_natsind_4
			Alexander_Manuilov_5
		}
		add_popularity = {
			ideology = communism
			popularity = 0.05
		}
		set_country_flag = RUS_kornilov_russia_38_kadet_government
	}
	option = {
		name = rcw_kcr.1.e #Переформировать из эсеров Савинкова
		if = {
			limit = { national_syndicalism > 0.05 }
			add_to_variable = { var = RUS_Kornilov_power value = 1 }
			else = {
				add_to_variable = { var = RUS_Kornilov_power value = -1 }
			}
		}
		add_popularity = {
			ideology = national_syndicalism
			popularity = 0.075
		}
		add_political_power = 50
		remove_ideas_with_trait = head_of_government
		remove_ideas_with_trait = foreign_minister
		remove_ideas_with_trait = economy_minister
		remove_ideas_with_trait = security_minister
		remove_ideas_with_trait = head_of_intel
		remove_ideas_with_trait = military_minister
		add_ideas = {
			Boris_Savinkov_natsind_1
			Sergey_Sazonov_2
			Pyotr_Heyden_3
			Boris_Savinkov_natsind_4
			Alexander_Krasilnikov_5
		}
		add_popularity = {
			ideology = communism
			popularity = 0.05
		}
		set_country_flag = RUS_kornilov_russia_38_savinkov_government
