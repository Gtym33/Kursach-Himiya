		}

	}

	}
	
	#Признать независимость

		name = finland.10.a
		every_state = {
			limit = { is_core_of = FIN  NOT = { state = 1049 state = 722 } }
			remove_core_of = RUS
			remove_core_of = RSS
			remove_core_of = CCA
		}
		every_country = {
			limit = {
				OR = {
					tag = ROOT
					is_in_faction_with = ROOT
					is_subject_of = ROOT
				}
			}
			if = {
				limit = {
					is_in_array = {
						array = unrecognized_states
						value = FIN.id
					}
				}
				FIN = {
					THIS_country_will_be_recognized_by_PREV = yes
				}
			}
		}
	}
	
	#Не признавать независимость
	option = {
		name = finland.10.b
	}
