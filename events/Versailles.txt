country_event = {
	id = versailles.100
	title = versailles.100.t
	desc = versailles.100.d
	picture = GFX_report_event_germany_civil_war_soldiers
	
	trigger = {
		has_global_flag = versailles_peace_treaty_in_progress
		NOT = { has_global_flag = WWI_is_now }
		has_country_flag = WWI_Entente_member
		is_faction_leader = yes
		has_global_flag = WWI_BUL_HAS_CAPITULATED
	}

	fire_only_once = yes
	
	mean_time_to_happen = {
		days = 90
	}
	hidden = yes
	
	option = {
		name = versailles.100.a
		BUL = { leave_faction = yes }
		hidden_effect = {
			if = {
				limit = { BUL = { NOT = { has_idea = BUL_interunion_commission_idea } } }
				set_variable = { global.Entente_major_conf_members = 0 }
				set_variable = { global.Entente_keep_white_peace_with_BUL = 0 }
				set_variable = { global.Entente_not_keep_white_peace_with_BUL = 0 }
				set_variable = { global.Entente_major_conf_votes = 0 }
				every_country = {
					limit = { has_country_flag = WWI_Entente_member is_subject = no NOT = { tag = BUL } }
					if = {
						limit = { OR = { tag = USA tag = RUS } }
						add_to_variable = { global.Entente_major_conf_members = 3 }
						set_variable = { vote_power = 3 }
					}
					else_if = {
						limit = { OR = { tag = FRA tag = ENG } }
						add_to_variable = { global.Entente_major_conf_members = 4 }
						set_variable = { vote_power = 4 }
					}
					else = {
						add_to_variable = { global.Entente_major_conf_members = 1 }
						set_variable = { vote_power = 1 }
					}
					country_event = { id = versailles.101 days = 12 }
				}
				else = {
					set_variable = { global.Entente_major_conf_members = 0 }
					set_variable = { global.Entente_Dobruja_ROM = 0 }
					set_variable = { global.Entente_Dobruja_not_fully_ROM = 0 }
					set_variable = { global.Entente_major_conf_votes = 0 }
					every_country = {
						limit = { has_country_flag = WWI_Entente_member is_subject = no NOT = { tag = BUL } }
						if = {
							limit = { OR = { tag = USA tag = RUS } }
							add_to_variable = { global.Entente_major_conf_members = 3 }
							set_variable = { vote_power = 3 }
						}
						else_if = {
							limit = { OR = { tag = FRA tag = ENG } }
							add_to_variable = { global.Entente_major_conf_members = 4 }
							set_variable = { vote_power = 4 }
						}
						else = {
							add_to_variable = { global.Entente_major_conf_members = 1 }
							set_variable = { vote_power = 1 }
						}
						country_event = { id = versailles.104 days = 12 }
					}
				}
			}
		}
	}
}

	option = { #Эти соглашения должны быть пересмотрены
		name = versailles.101.a
		add_to_variable = { global.Entente_not_keep_white_peace_with_BUL = vote_power }
		
		add_to_variable = { global.Entente_major_conf_votes = vote_power }
		
		hidden_effect = {
			if = {
				limit = { check_variable = { global.Entente_major_conf_votes = global.Entente_major_conf_members } }

			}
		}

			factor = 2
		}
