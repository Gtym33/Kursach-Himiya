		name = rcw_bfk.9.a

			start_civil_war = {
				ideology = communism
				size = 0.3
				capital = 915

			}
		}
		every_state = {
			limit = {
				is_owned_by = EST
				is_controlled_by = RUS
			}
			every_country = {
				limit = { original_tag = EST has_government = communism }
				transfer_state = PREV
			}
		}
		every_country = {
			limit = { original_tag = EST has_government = communism }
			set_cosmetic_tag = EST_labor_commune
		}
		if = {
			limit = { has_war_with = NWA }
			every_country = {
				limit = { original_tag = EST has_government = communism }
				declare_war_on = {
					target = NWA
					type = annex_everything
				}
			}
		}
		if = {
			limit = { has_war_with = WVA }
			every_country = {
				limit = { original_tag = EST has_government = communism }

					target = WVA
					type = annex_everything
				}
			}
		}

			factor = 999
		}
		every_country = {
			limit = { original_tag = EST has_government = communism }

		}

		name = rcw_bfk.9.b

		}
