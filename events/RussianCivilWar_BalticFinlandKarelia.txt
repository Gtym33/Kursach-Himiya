	}

								is_core_of = EST
							}
						}
						CONTROLLER = { OR = { tag = ROOT is_subject_of = ROOT is_in_faction_with = ROOT } }
					}
					event_target:EST_CW = { set_state_owner = PREV }
				}
				set_politics = {
					ruling_party = communism
					election_frequency = 36
					elections_allowed = no
				}
				set_popularities = {
					communism = 100
				}
			}
			declare_war_on = {
				target = EST
				type = annex_everything
			}
			add_civil_war_target = EST
			set_cosmetic_tag = EST_labor_commune
			every_country = {
				limit = { has_offensive_war_with = ROOT NOT = { has_war_with = event_target:EST_CW } }
				declare_war_on = {
					target = event_target:EST_CW
					type = annex_everything
				}
			}
			every_country = {
				limit = { has_defensive_war_with = ROOT NOT = { has_war_with = event_target:EST_CW } }
				event_target:EST_CW = {
					declare_war_on = {
						target = PREV
						type = annex_everything
					}
				}
			}
			ROOT = { event_target:EST_CW = { subjecting_script_with_transfer_parties_dominion = yes } }
			hidden_effect = {
				every_state = {
					limit = {
						OR = {
							is_owned_by = EST
							is_owned_by = event_target:EST_CW
							AND = {
								OWNER = { OR = { tag = ROOT is_subject_of = ROOT } }
								is_core_of = EST
							}
						}
					}
					set_state_province_controller = {
						controller = event_target:EST_CW
						limit = { OR = { tag = ROOT is_subject_of = ROOT is_in_faction_with = ROOT } }
					}
				}
			}
			hidden_effect = {
				every_country = {
					limit = {
						NOT = {
							tag = PREV
							is_any_former_russian_empire_country = yes
							former_russian_empire_baltic_states_country = yes tag = HRB
						}
					}
					set_THIS_country_non_recognize_PREV = yes
				}
			}
		}
		EST = {
			transfer_units_fraction= {
				target = event_target:EST_CW
				size = 0.15
				stockpile_ratio = 0.15
				army_ratio = 0.15
				navy_ratio = 0
				air_ratio = 0
				keep_unit_leaders_trigger = { always = yes }
			}
		}
		clear_global_event_target = EST_CW
		ai_chance = {
			factor = 999
		}
	}
	option = {
		name = rcw_bfk.9.b
		ai_chance = {
			factor = 0
		}
	}
