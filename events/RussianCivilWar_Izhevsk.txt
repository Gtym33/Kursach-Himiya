			}

		modifier = {
			factor = 0.05
			OR = {

		name = rcw_izhevsk.1.a

			value = 1

			transfer_state = 399

				every_state = {

			country_event = { id = rcw_izhevsk.3 days = 14 }
			country_event = { id = rcw_izhevsk.5 days = 30 }
			hidden_effect = {
				every_country = {
					limit = {
						NOT = {
							tag = PREV
							is_any_former_russian_empire_country = yes
							former_russian_empire_baltic_states_country = yes tag = HRB
						}
					}
					set_THIS_country_non_recognize_PREV = yes
				}
			}

				target = ROOT

			}

	}

				}

	}

		if = {
			limit = { has_idea = NVA_decentralization_of_army_a }
			remove_ideas = NVA_decentralization_of_army_a
		}
		if = {
			limit = { has_idea = IZH_local_peoples_armies_idea }
