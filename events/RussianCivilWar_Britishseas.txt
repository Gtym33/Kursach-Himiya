# Татарские сепаратисты
country_event = {
	id = rcw_britishseas.1
	title = rcw_britishseas.1.t
	desc = rcw_britishseas.1.d
	picture = GFX_report_event_rcw_czech_uprising
	
	is_triggered_only = yes
	
	option = {
		name = rcw_britishseas.1.a
		
		hidden_effect = {
			every_country = {
				limit = { has_country_flag = WWI_post_german_russia_collaborators_new_country_BSH_TAR_CHV }
				clr_country_flag = WWI_post_german_russia_collaborators_new_country_BSH_TAR_CHV
			}
		}
		ai_chance = {
			factor = 0
		}
	}
	option = {
		name = rcw_britishseas.1.b
		if = {
			limit = { BSH = { has_country_flag = WWI_post_german_russia_collaborators_new_country_BSH_TAR_CHV } }
			declare_war_on = { target = BSH type = annex_everything }
		}
		if = {
			limit = { TAR = { has_country_flag = WWI_post_german_russia_collaborators_new_country_BSH_TAR_CHV } }
			declare_war_on = { target = TAR type = annex_everything }
		}
		if = {
			limit = { CHV = { has_country_flag = WWI_post_german_russia_collaborators_new_country_BSH_TAR_CHV } }
			declare_war_on = { target = CHV type = annex_everything }
		}
		
		hidden_effect = {
			every_country = {
				limit = { has_country_flag = WWI_post_german_russia_collaborators_new_country_BSH_TAR_CHV }
				clr_country_flag = WWI_post_german_russia_collaborators_new_country_BSH_TAR_CHV
			}
		}
		ai_chance = {
			factor = 1
		}
	}
}

country_event = {
	id = rcw_britishseas.6
	title = rcw_britishseas.6.t
	desc = rcw_britishseas.6.desc
	picture = GFX_report_event_rcw_czech_uprising
	
	is_triggered_only = yes
	
	option = {
		name = rcw_britishseas.6.a
		if = {
			limit = { GER = { is_ai = yes } PRE = { is_ai = no } }
			GER = { change_tag_from = PRE }
		}
		PRE = {
			every_unit_leader = {
				set_nationality = GER
			}
		}
		GER = {
			annex_country = {
				target = PRE
				transfer_troops = yes
			}
		}
		ai_chance = {
			factor = 0
		}
	}
	option = {
		trigger = { PRE = { any_neighbor_country = { OR = { tag = ROOT is_subject_of = ROOT is_in_faction_with = ROOT has_war_with = ROOT } } } }
		name = rcw_britishseas.6.b
		declare_war_on = { target = PRE type = annex_everything }
		PRE = { country_event = rcw_britishseas.8 }
		ai_chance = {
			factor = 1
		}
	}
}
