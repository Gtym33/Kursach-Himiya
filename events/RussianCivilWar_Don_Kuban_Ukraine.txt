country_event = {
	id = rcw_dku.-1
	hidden = yes
	
	trigger = {
		always = no
		tag = RVA
		OR = {
			AND = {
				has_idea = RVA_KUB_rada_1
				234 = { is_capital = yes }
				num_of_controlled_states < 10
			}
			AND = {
				has_idea = RVA_KUB_rada_2
				num_of_controlled_states < 10
			}
		}
	}
	
	mean_time_to_happen = {
		days = 0
	}
	
	option = {
		if = {
			limit = { has_idea = RVA_KUB_rada_1 }
			swap_ideas = {
				remove_idea = RVA_KUB_rada_1
				add_idea = RVA_KUB_rada
			}
		}
		else_if = {
			limit = { has_idea = RVA_KUB_rada_2 }
			swap_ideas = {
				remove_idea = RVA_KUB_rada_2
				add_idea = RVA_KUB_rada_1
			}
		}
	}
}
