country_event = {
	id = rcw_amrprm.1
	title = rcw_amrprm.1.t
	desc = rcw_amrprm.1.desc

	picture = GFX_report_event_amur_blagoveshensk

	trigger = {
		has_start_date < 1917.12.1 #примерно через месяц после революции
		tag = RUS
		NOT = { country_exists = AMR }
		RUS = {
			NOT = { has_country_flag = AMR_revolt }
			has_country_flag = RUS_Oktober_revolution_fired
		}
		1039 = { OWNER = { OR = { is_subject_of = ROOT tag = ROOT tag = FER } } }
	}
	
	mean_time_to_happen = {
		days = 30
	}
	
	fire_only_once = yes

	immediate = {
		hidden_effect = {
			if = {
				limit = { FER = { is_ai = no exists = yes } }
				FER = { country_event = rcw_amrprm.0 }
			}
			every_unit_leader = { limit = { OR = { has_id = 64403 has_id = 64404 has_id = 64405 } } set_nationality = AMR }
			AMR = {
				inherit_technology = RUS
				set_cosmetic_tag = AMR_oblast
				transfer_state = 561
				transfer_state = 1024
				transfer_state = 1039
				transfer_state = 1184
				add_ideas = neutral_foreign_policy_law
				add_ideas = RUS_red_guards_provisional_government
				add_ideas = AMR_dvoevlastie
				remove_all_foreing_locked_templates_from_liberated_country = yes
				hidden_effect = {
					every_country = {
						limit = {
							NOT = {
								tag = PREV
								is_any_former_russian_empire_country = yes
								former_russian_empire_baltic_states_country = yes tag = HRB
							}
						}
						set_THIS_country_non_recognize_PREV = yes
					}
				}
			}
		}
	}
	
	option = {
		name = rcw_amrprm.1.a
		hidden_effect = {
			AMR = {
				load_oob = "AMR_1917"
				country_event = { id = rcw_amrprm.2 days = 60 }
			}
		}
	}
	
	option = {
		name = rcw_amrprm.1.b
		trigger = {
			is_ai = no
			AMR = { OR = { exists = no is_ai = yes } }
			FER = { OR = { exists = no is_ai = yes } }
		}
		hidden_effect = {
			AMR = {
				load_oob = "AMR_1917"
				country_event = { id = rcw_amrprm.2 days = 60 }
			}
		}
		AMR = { change_tag_from = RUS }
	}

}
