		}

			factor = 1
		}
	}

		if = {

				FROM = {
					any_owned_state = {
						NOT = { state = 104 state = 804 state = 45 state = 1374 state = 1449 state = 1450 }
					}
				}
			}
			FROM = {
				every_owned_state = {
					limit = {

				}
			}
			else = {
				annex_country = {
					target = FROM
					transfer_troops = no
				}
			}
		}
	}
}
