#Деятельность Савинкова
country_event = {
	id = rcw_nva.4
	title = rcw_nva.4.t
	desc = rcw_nva.4.desc

	is_triggered_only = yes
	
	option = {
		trigger = { DON = { has_idea = DON_Volunteer_army_presence_idea } NOT = { country_exists = RVA } }
		name = rcw_nva.4.a
		if = {
			limit = { has_global_flag = WWI_is_now }
			DON = { country_event = { id = rcw_nva.5 days = 20 } }
		}
		if = {
			limit = { NOT = { has_global_flag = WWI_is_now } }
			random_list = {
				60 = { DON = { country_event = { id = rcw_nva.5 days = 20 } } }
				40 = { }
			}
		}
	}
	
	option = {
		trigger = { DON = { NOT = { has_idea = DON_Volunteer_army_presence_idea } } country_exists = RVA }
		name = rcw_nva.4.b
		if = {
			limit = { has_global_flag = WWI_is_now }
			RVA = { country_event = { id = rcw_nva.5 days = 20 } }
		}
		if = {
			limit = { NOT = { has_global_flag = WWI_is_now } }
			random_list = {
				60 = { RVA = { country_event = { id = rcw_nva.5 days = 20 } } }
				40 = { }
			}
		}
	}
}
