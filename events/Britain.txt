add_namespace = britain

#Ирландское сопротивление - февральская революция
country_event = {
	id = britain.1
	hidden = yes
	
	trigger = {
		has_start_date < 1917.3.8
		tag = ENG
		RUS = { NOT = { has_country_flag = RUS_monarchy_not_fall } }
	}
	
	fire_only_once = yes
	option = {
		every_owned_state = {
			limit = { is_core_of = IRE NOT = { is_core_of = ROOT } }
			add_resistance = 18
			add_resistance_target = {
				amount = 18
				occupied = IRE
			}
		}
	}
}
