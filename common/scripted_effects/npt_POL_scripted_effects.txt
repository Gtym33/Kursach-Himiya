POL_polska_polushaet_oshko = {
	add_to_variable = { POL_oshko = POL_osch_pol }
	clamp_variable = {
		var = POL_oshko
		min = 0
		max = 1
	}
}

POL_germ_polushaet_oshko = {
	add_to_variable = { POL_oshko_ger = POL_osch_ger }
	clamp_variable = {
		var = POL_oshko_ger
		min = 0
		max = 1
	}
}
POL_reset_oshko = {
	set_temp_variable = { POL_osch_pol = 1 }
	set_temp_variable = { POL_oshko = 1 }
	set_temp_variable = { POL_osch_ger = 1 }
	set_temp_variable = { POL_oshko_ger = 1 }
}
