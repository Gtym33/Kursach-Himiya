RGR_kolichestvo_mission_ataman_plus = {
	add_to_variable = { RGR_ataman = RGR_ataman_plus }
	clamp_variable = {
		var = RGR_ataman
		min = 0
		max = 1
	}
}

RGR_kolichestvo_mission_ataman_minus = {
	add_to_variable = { RGR_ataman = RGR_ataman_minus }
	clamp_variable = {
		var = RGR_ataman
		min = 0
		max = 1
	}
}
RGR_kolichestvo_mission_ataman_reset = {
	set_temp_variable = { RGR_ataman_plus = -1 }
	set_temp_variable = { RGR_ataman_minus = -1 }
	set_temp_variable = { RGR_ataman = 1 }
}
