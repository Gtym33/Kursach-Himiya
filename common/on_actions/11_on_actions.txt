on_actions = {
on_ruling_party_change = { # Note: the on action only runs after the game's start
        effect = {
            if = {
                limit = {
                    original_tag = PRM
                    has_government = national_syndicalism
                }
                set_cosmetic_tag = PRM_nat
            }
            else_if = {
                limit = {
                    original_tag = PRM
                }
                drop_cosmetic_tag = yes
            }
        }
    }

on_ruling_party_change = { # Note: the on action only runs after the game's start
        effect = {
            if = {
                limit = {
                    original_tag = TRK
                    has_government = communism
                }
                set_cosmetic_tag = TRK_K
            }
            else_if = {
                limit = {
                    original_tag = TRK
                }
                drop_cosmetic_tag = yes
            }
        }
    }
}
